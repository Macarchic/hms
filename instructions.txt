  source .venv/bin/activate
                                                                                                                                                                               
  # спочатку перевір на 100 файлах (~10 сек)                                                                                                                                   
  python src/precompute.py --limit 100

  python src/precompute.py

  Крок 2 — тренування
                                                                                                                                                                               
  # EEG модель (EfficientNet-B0 на STFT спектрограмах)
  python src/train.py --modality eeg --epochs 30 --batch_size 16                                                                                                               
                  
  # Спектрограмна модель (EfficientNet-B0 на готових спектрограмах Kaggle)                                                                                                     
  python src/train.py --modality spec --epochs 30 --batch_size 16
                                                                                                                                                                               
  # Санітарна перевірка (1 епоха, 8 семплів)
